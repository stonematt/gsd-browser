```javascript
function getPhaseDepth(numStr) {
  return String(numStr).split('.').length - 1; // "4"->0, "4.5"->1, "4.5.1"->2
}

function getImmediateParent(numStr) {
  const parts = String(numStr).split('.');
  if (parts.length <= 1) return null;
  return parts.slice(0, -1).join('.');
}

function comparePhaseNums(a, b) {
  const partsA = String(a).split('.').map(Number);
  const partsB = String(b).split('.').map(Number);
  const len = Math.max(partsA.length, partsB.length);
  for (let i = 0; i < len; i++) {
    const diff = (partsA[i] || 0) - (partsB[i] || 0);
    if (diff !== 0) return diff;
  }
  return 0;
}

function buildPhaseTree(phases) {
  const byKey = {};
  const roots = [];
  for (const phase of phases) {
    byKey[String(phase.numStr)] = { ...phase, children: [] };
  }
  for (const phase of phases) {
    const parentNum = getImmediateParent(phase.numStr);
    if (parentNum && byKey[parentNum]) {
      byKey[parentNum].children.push(byKey[String(phase.numStr)]);
    } else {
      roots.push(byKey[String(phase.numStr)]);
    }
  }
  return roots;
}

function splitHistoryFrontier(phases) {
  // phases: flat sorted array
  let boundaryIdx = phases.length;
  for (let i = 0; i < phases.length; i++) {
    if (phases[i].status !== 'complete') {
      boundaryIdx = i;
      break;
    }
  }
  return { history: phases.slice(0, boundaryIdx), frontier: phases.slice(boundaryIdx) };
}

function rectDimensions(planCount, reqCount, scale) {
  const minW = 12, minH = 12;
  const pxPerPlan = 8, pxPerReq = 6;  // starting points — visual iteration in Task 2
  const w = Math.max(minW, (planCount || 0) * pxPerPlan) * scale;
  const h = Math.max(minH, (reqCount || 0) * pxPerReq) * scale;
  return { width: Math.round(w), height: Math.round(h) };
}
```
