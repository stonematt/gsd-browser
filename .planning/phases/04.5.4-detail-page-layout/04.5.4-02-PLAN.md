**New `groupPhaseFiles(mdFiles)` function:**
```javascript
function groupPhaseFiles(files) {
  const planNums = [...new Set(
    files.filter(f => f.includes('-PLAN.md'))
         .map(f => f.match(/^[\d.]+-(\d+)-PLAN\.md/)?.[0] ? f.match(/^([\d.]+-\d+)/)?.[1] : null)
         .filter(Boolean)
  )].sort();
  const above = [];
  for (const num of planNums) {
    const plan = files.find(f => f.startsWith(num + '-PLAN'));
    const summary = files.find(f => f.startsWith(num + '-SUMMARY'));
    if (plan) above.push(plan);
    if (summary) above.push(summary);
  }
  const aboveSet = new Set(above);
  const below = files.filter(f => f.endsWith('.md') && !aboveSet.has(f));
  return { above, below };
}
```
