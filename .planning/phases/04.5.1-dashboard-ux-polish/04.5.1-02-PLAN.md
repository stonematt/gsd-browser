5. Add helper functions INSIDE the script block (not global):
   ```javascript
   function isSubPhase(numStr) {
     return numStr.includes('.');
   }
   function getParentPhaseNum(numStr) {
     return numStr.split('.')[0];
   }
   function groupPhases(phases) {
     const groups = [];
     for (const phase of phases) {
       if (!isSubPhase(phase.numStr)) {
         groups.push({ parent: phase, children: [] });
       } else {
         const parentNum = getParentPhaseNum(phase.numStr);
         const group = groups.find(g => g.parent.numStr === parentNum);
         if (group) {
           group.children.push(phase);
         } else {
           // Orphaned sub-phase — render as peer (fallback)
           groups.push({ parent: phase, children: [] });
         }
       }
     }
     return groups;
   }
   ```
