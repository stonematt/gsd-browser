## Summary

Phase 4.5 builds a multi-view SPA that replaces `public/index.html` as the app entry point. The dashboard aggregates GSD planning artifacts across all registered sources, surfacing project health at a glance. Three views are needed: dashboard (cards), project detail (phase timeline + docs), and the existing file browser (preserved as a view).

All required server-side capabilities either already exist or can be added with minimal new infrastructure. STATE.md parsing works with a simple two-level YAML regex parser — no external dependency needed, verified against the actual STATE.md format. Phase status detection (complete/in-progress/pending) is derived directly from file presence patterns in `.planning/phases/*/` directories. Branch-aware operations use `execFile` with `git -C <repoPath>` — no shell injection risk, verified working.

The SPA expands existing `public/index.html` hash routing with view prefixes (`#/project/name`, `#/browse/name/path`). The dashboard view replaces the current default. The browse view preserves the existing three-panel layout. This single-file SPA pattern (inline style/script, no build tooling) is already established in `sources.html` and `index.html`.

**Primary recommendation:** Implement three new Fastify API endpoints (`/api/dashboard`, `/api/projects/:name/detail`, `/api/projects/:name/branches`), a server-side `parseStateMd()` utility, and replace `public/index.html` with a multi-view SPA that shares the dark theme and reuses `/render?fragment=true`.

### Pattern 1: STATE.md Frontmatter Parser (server-side)

Parses `---\n...\n---` YAML block at top of file. Handles two-level nesting (top-level keys + `progress:` block). No external dependency.

```javascript
// Source: verified against actual STATE.md in gsd-browser repo
'use strict';

/**
 * Parse GSD STATE.md frontmatter.
 * Returns null if no frontmatter found.
 * @param {string} content - File content
 * @returns {{ status, last_activity, last_updated, milestone, progress: { percent, total_phases, completed_phases, total_plans, completed_plans } } | null}
 */
function parseStateMd(content) {
  const match = content.match(/^---\n([\s\S]*?)\n---/);
  if (!match) return null;

  const yaml = match[1];
  const lines = yaml.split('\n');
  const result = {};
  let currentKey = null;

  for (const line of lines) {
    if (line.match(/^\s+/)) {
      // Indented sub-key
      const m = line.match(/^\s+(\w+):\s*(.+)/);
      if (m && currentKey) {
        if (typeof result[currentKey] !== 'object') result[currentKey] = {};
        result[currentKey][m[1]] = m[2].trim();
      }
    } else {
      const m = line.match(/^(\w[\w_]*):\s*(.*)/);
      if (m) {
        currentKey = m[1];
        const val = m[2].replace(/^["']|["']$/g, '');
        result[currentKey] = val || {};
      }
    }
  }
  return result;
}
```

**Confidence:** HIGH — tested against actual STATE.md. Handles top-level strings, quoted strings, and the nested `progress:` block.

### Pattern 2: Phase Status Detection (server-side)

Determines phase status by scanning file presence in a phase directory. No parsing of file content needed.

```javascript
// Source: verified against all 5 phase directories in gsd-browser
async function getPhaseInfo(phaseDirPath) {
  let entries;
  try {
    entries = await fs.readdir(phaseDirPath);
  } catch {
    return null;
  }

  const plans = entries.filter(f => /\d+-\d+-PLAN\.md$/.test(f));
  const summaries = entries.filter(f => /\d+-\d+-SUMMARY\.md$/.test(f));
  const hasVerification = entries.some(f => f.endsWith('-VERIFICATION.md'));

  let status;
  if (hasVerification || (plans.length > 0 && summaries.length >= plans.length)) {
    status = 'complete';
  } else if (plans.length > 0) {
    status = 'in-progress';
  } else {
    status = 'pending';
  }

  return {
    planCount: plans.length,
    completedPlans: summaries.length,
    status,
    files: entries
  };
}
```

**Confidence:** HIGH — verified against 01-foundation (complete), 04-browser-ui (complete), 04.5-gsd-dashboard (pending).

### Pattern 3: Phase Directory Name Parsing (server-side)

// Source: verified against all phase directory names
function parsePhaseDir(dirName) {
  const m = dirName.match(/^(\d+(?:\.\d+)?)-(.+)$/);
  if (!m) return null;
  return {
    num: parseFloat(m[1]),
    numStr: m[1],
    slug: m[2],
    dir: dirName

}

### Pattern 4: Git Operations (server-side)

Use `execFile` (not `exec`) — args passed as array, no shell injection possible. Always wrap in try/catch and return null on failure.

```javascript
// Source: verified with actual gsd-browser repo
const { execFile } = require('node:child_process');
const { promisify } = require('node:util');
const execFileAsync = promisify(execFile);

// Branch name validation — reject anything with shell metacharacters
function isValidBranchName(name) {
  return /^[a-zA-Z0-9._\-/]+$/.test(name);
}

async function getBranchesWithPlanning(repoPath) {
  try {
    const { stdout } = await execFileAsync('git', ['-C', repoPath, 'branch', '--format=%(refname:short)']);
    const branches = stdout.split('\n').filter(Boolean);
    const withPlanning = [];
    for (const branch of branches) {
      try {
        await execFileAsync('git', ['-C', repoPath, 'cat-file', '-e', `${branch}:.planning/STATE.md`]);
        withPlanning.push(branch);
      } catch { /* not found on this branch */ }
    }
    return withPlanning;
  } catch {
    return []; // Not a repo or git unavailable
  }
}

async function readFileFromBranch(repoPath, branch, filePath) {
  if (!isValidBranchName(branch)) return null;
  try {
    const { stdout } = await execFileAsync(
      'git', ['-C', repoPath, 'show', `${branch}:${filePath}`],
      { maxBuffer: 1024 * 1024 }
    );
    return stdout;
  } catch {
    return null;
  }
}

async function listDirFromBranch(repoPath, branch, dirPath) {
  if (!isValidBranchName(branch)) return [];
  try {
    const { stdout } = await execFileAsync('git', ['-C', repoPath, 'ls-tree', '--name-only', `${branch}:${dirPath}`]);
    return stdout.split('\n').filter(Boolean);
  } catch {
    return [];
  }
}
```

**Confidence:** HIGH — all commands verified against gsd-browser repo. Injection protection confirmed with test.

### Pattern 5: Hash-Based Multi-View Routing (client-side SPA)

Extends the existing hash routing in `index.html`. View is determined by the first path segment after `#/`.

```javascript
// Source: extends existing parseHash() pattern from index.html
function parseAppHash(hash) {
  if (!hash || hash === '#' || hash === '#/') {
    return { view: 'dashboard' };
  }
  const raw = hash.replace(/^#\/?/, '');
  const parts = raw.split('/');

  if (parts[0] === 'project') {
    return {
      view: 'project',
      sourceName: decodeURIComponent(parts[1] || ''),
      branch: parts[2] ? decodeURIComponent(parts[2]) : null
    };
  }
  if (parts[0] === 'browse') {
    const sourceName = decodeURIComponent(parts[1] || '');
    const filePath = parts.slice(2).map(decodeURIComponent).join('/');
    return { view: 'browse', sourceName, filePath };
  }
  return { view: 'dashboard' };
}
```

| Hash | View |
|---|---|
| `#` or `#/` | dashboard |
| `#/project/gsd-browser` | project detail for gsd-browser (default branch) |
| `#/project/gsd-browser/feature` | project detail for gsd-browser on `feature` branch |
| `#/browse/gsd-browser/.planning/STATE.md` | file browser for that file |

**Confidence:** HIGH — tested with all expected navigation paths.

### Anti-Patterns to Avoid

- **Using `exec()` for git commands:** Shell injection risk. Always use `execFile()` with args array.
- **Using `js-yaml` or `gray-matter` npm packages:** Both are ESM-only and conflict with CJS codebase. The two-level regex parser is sufficient.
- **Adding a build step / bundler:** The project deliberately has no build tooling. Single-file SPA with inline script.
- **Blocking the event loop on git ops:** All git operations are async via `execFileAsync`. Never use synchronous child_process calls for git.
- **Using `child_process.exec` with template strings:** Allows injection. Use `execFile` with array args.
- **Separate HTML files for dashboard vs project views:** The existing pattern is single-file SPA (index.html handles all views). Stay consistent.

## Common Pitfalls

### Pitfall 1: Git `exec` vs `execFile`
**What goes wrong:** Using `child_process.exec('git show ' + branch + ':path')` allows shell injection if branch name contains metacharacters (`; cat /etc/passwd`).
**Why it happens:** `exec()` runs through a shell; `execFile()` does not.
**How to avoid:** Always use `execFile()` with args array. Validate branch names with `/^[a-zA-Z0-9._\-/]+$/` before use.
**Warning signs:** Any git call that builds a command string with string interpolation.

### Pitfall 2: MANAGEMENT_CSP Required on New API Routes
**What goes wrong:** New `/api/dashboard` and `/api/projects/*` routes use the global preHandler CSP (`script-src 'none'`), which blocks `fetch()` calls from SPA pages that have inline scripts.
**Why it happens:** The global preHandler sets strict CSP; management routes must override with `MANAGEMENT_CSP` via `onSend` hook.
**How to avoid:** Add `onSend` hook to all new API routes, same as existing `/api/sources` routes.
**Warning signs:** Fetch calls from the dashboard SPA returning 200 but being blocked by browser CSP console errors.

### Pitfall 3: `activeSources` vs `listSources()` in New Routes
**What goes wrong:** New dashboard endpoints call `listSources()` directly (reads from disk each time), while tree endpoint uses `activeSources` (server closure). Inconsistency means add/remove won't reflect immediately in dashboard.
**Why it happens:** Dashboard needs enriched source data that includes `available`, `conventions` fields.
**How to avoid:** Use `listSources()` in `/api/dashboard` (same pattern as `/api/sources`). This is correct — dashboard needs fresh config data. Use `activeSources` for path-gated routes like `/file` and `/render`.

### Pitfall 4: Phase Directory Sorting with Decimal Phase Numbers
**What goes wrong:** `readdir()` returns directory names in filesystem order. Sorting by string (`04.5 < 05`) works, but only if zero-padded consistently. Sorting by string with `localeCompare` can fail for multi-digit phases.
**Why it happens:** `"10" < "2"` lexicographically; decimal phases like `04.5` need numeric sort.
**How to avoid:** Parse phase number with `parseFloat()` after extracting the numeric prefix. Sort by numeric value. Confirmed: `04.5` parses to `4.5`, `05` parses to `5` — correct ordering.

### Pitfall 5: Quick-Link Files May Not Exist
**What goes wrong:** Not all GSD projects have all three quick-link files (PROJECT.md, STATE.md, ROADMAP.md). Linking to a file that doesn't exist shows a 404 in the render pane.
**Why it happens:** Assuming all GSD conventions are present.
**How to avoid:** Check file existence server-side when building dashboard data. Include `"exists": true/false` in quickLinks array. Client shows link only when `exists: true`.

### Pitfall 6: Branch Discovery at Dashboard Load is Too Slow
**What goes wrong:** Running `git branch` + `git cat-file` for every source at dashboard load time blocks the page render, especially with many sources.
**Why it happens:** Branch discovery requires per-branch git operations.
**How to avoid:** Branch discovery is on-demand — only in `/api/projects/:name/detail`, not in `/api/dashboard`. Dashboard card shows branch count badge only after the detail page triggers `/api/projects/:name/branches`. This is a locked decision from CONTEXT.md.

### Pitfall 7: `@fastify/static` Prefix Collision with New Routes
**What goes wrong:** New routes like `/api/dashboard` or `/api/projects/` might conflict with static file serving if registered in wrong order.
**Why it happens:** `@fastify/static` is registered with `prefix: '/'`.
**How to avoid:** Register API routes explicitly before static serving takes over. Fastify matches explicit routes before static fallback — this already works in the codebase. No change needed.

### Dashboard API Endpoint (server.js addition)

```javascript
// GET /api/dashboard — aggregate GSD project data for all registered sources
fastify.get('/api/dashboard', {
  onSend: async (request, reply, payload) => {
    reply.header('Content-Security-Policy', MANAGEMENT_CSP);
    return payload;
  }
}, async (request, reply) => {
  const allSources = await listSources();
  const projects = [];
  const other = [];

  for (const source of allSources) {
    if (!source.available) continue;

    // Check for GSD artifact
    const statePath = path.join(source.path, '.planning', 'STATE.md');
    let stateContent = null;
    try {
      stateContent = await fs.readFile(statePath, 'utf8');
    } catch { /* not a GSD project */ }

    if (!stateContent) {
      other.push({ name: source.name, path: source.path });
      continue;
    }

    const state = parseStateMd(stateContent);
    const phases = await buildPhaseList(source.path);
    const quickLinks = await buildQuickLinks(source.path);

    projects.push({
      name: source.name,
      path: source.path,
      isGsd: true,
      progress: {
        percent: parseInt(state?.progress?.percent ?? 0),
        completed_phases: parseInt(state?.progress?.completed_phases ?? 0),
        total_phases: parseInt(state?.progress?.total_phases ?? 0),
      },
      currentFocus: state?.status ?? '',
      lastActivity: state?.last_activity ?? '',
      milestone: state?.milestone ?? '',
      quickLinks,
      phaseStatus: phases
    });
  }

  return reply.send({ projects, other });
});
```

### Phase List Builder (server.js helper)

```javascript
async function buildPhaseList(sourcePath) {
  const phasesDir = path.join(sourcePath, '.planning', 'phases');
  let dirs;
  try {
    dirs = await fs.readdir(phasesDir);
  } catch {
    return [];
  }

  const phases = [];
  for (const dir of dirs) {
    const parsed = parsePhaseDir(dir);
    if (!parsed) continue;
    const info = await getPhaseInfo(path.join(phasesDir, dir));
    if (!info) continue;
    phases.push({
      num: parsed.num,
      numStr: parsed.numStr,
      slug: parsed.slug,
      status: info.status,
      planCount: info.planCount,
      completedPlans: info.completedPlans
    });
  }

  // Sort by numeric phase number (handles 04.5 correctly)
  phases.sort((a, b) => a.num - b.num);
  return phases;
}
```

   - What we know: STATE.md frontmatter has `progress.percent` (e.g., `43`). Phase directory scanning provides independent phase-level status.
   - What's unclear: Should `/api/dashboard` trust the `percent` from STATE.md (which GSD agents write), or compute it from phase status files?
   - Recommendation: Trust STATE.md `percent` for the card display (it's the authoritative GSD artifact). Use phase file scanning for the per-phase dot colors (more granular, always current).

2. **"In-progress" vs "active" phase determination**
   - What we know: A phase with PLAN.md files but no VERIFICATION.md is `in-progress`. STATE.md `stopped_at` field names the current phase.
   - What's unclear: If `stopped_at` says "Phase 4.5 context gathered", does that mean 4.5 is active or that it's next?
   - Recommendation: Use filesystem evidence (phase with plans but no verification) for dot color. The `stopped_at` field is editorial context, not machine status.

3. **Single-branch repos: show branch badge?**
   - Marked as Claude's Discretion in CONTEXT.md.
   - Recommendation: Only show branch badge when branch count > 1. Single-branch repos are the common case; no UI clutter needed.
